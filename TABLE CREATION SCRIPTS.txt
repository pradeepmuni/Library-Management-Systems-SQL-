CREATE TABLE BOOKS
   (	ISBN10 CHAR(10 BYTE) NOT NULL UNIQUE, 
	TITLE VARCHAR2(3200 BYTE), 
	ISBN13 CHAR(13 BYTE) NOT NULL, 
	COVER VARCHAR2(300 BYTE), 
	PUBLISHER VARCHAR2(300 BYTE), 
	PAGES NUMBER, 
	 CONSTRAINT BCPK PRIMARY KEY (ISBN13)
   );


   CREATE TABLE BORROWER
   (	 CARD_NO CHAR(8 BYTE) NOT NULL, 
	SSN NUMBER(9,0), 
	FIRST_NAME VARCHAR2(50 BYTE), 
	LAST_NAME VARCHAR2(50 BYTE), 
	EMAIL VARCHAR2(50 BYTE), 
	ADDRESS VARCHAR2(50 BYTE), 
	CITY VARCHAR2(50 BYTE), 
	STATE CHAR(2 BYTE), 
	PHONE NUMBER(10,0), 
	CONSTRAINT BOPK PRIMARY KEY (CARD_NO)
  ) ;
  
  CREATE TABLE BRANCH
   (	BRANCH_ID NUMBER(38,0) NOT NULL, 
	BRANCH_NAME VARCHAR2(50 BYTE), 
	ADDRESS VARCHAR2(50 BYTE), 
	CONSTRAINT BRANCHPK PRIMARY KEY (BRANCH_ID)
);

  CREATE TABLE AUTHORS
   (	AUTHOR_ID NUMBER(*,0) NOT NULL, 
	AUTHOR_NAME VARCHAR2(50 BYTE), 
	 CONSTRAINT AUTHPK PRIMARY KEY (AUTHOR_ID)
 ) ;
 
  CREATE TABLE BOOK_AUTHORS
   (	AUTHOR_ID NUMBER(*,0) NOT NULL, 
	ISBN13 CHAR(13 BYTE) NOT NULL, 
	 CONSTRAINT BOOKAUTHPK PRIMARY KEY (AUTHOR_ID, ISBN13),
 	 CONSTRAINT AUTHFK FOREIGN KEY (AUTHOR_ID)
	  REFERENCES AUTHORS (AUTHOR_ID), 
	 CONSTRAINT BAFK FOREIGN KEY (ISBN13)
	  REFERENCES BOOKS(ISBN13) 
);

  CREATE TABLE BOOKCOPIES 
   (	BOOK_ID NUMBER(*,0) NOT NULL, 
	ISBN10 CHAR(10 BYTE) NOT NULL, 
	BRANCH_ID NUMBER(*,0), 
	NUM_COPY NUMBER(*,0), 
	 CONSTRAINT BOOKCOPIES_PK PRIMARY KEY (BOOK_ID),
  	 CONSTRAINT BOOKCOPIES_FK FOREIGN KEY (BRANCH_ID)
	  REFERENCES BRANCH (BRANCH_ID) , 
	 CONSTRAINT BVFKK FOREIGN KEY (ISBN10)
	  REFERENCES BOOKS (ISBN10) 
   ) ;

 CREATE TABLE BOOK_LOANS
   (	LOAN_ID NUMBER(*,0) NOT NULL, 
	ISBN10 CHAR(10 BYTE), 
	CARD_NO CHAR(10 BYTE), 
	DATE_OUT DATE, 
	DUE_DATE DATE, 
	DATE_IN DATE, 
	CONSTRAINT BOOKLOANS_PK PRIMARY KEY (LOAN_ID),
  	 CONSTRAINT CARDNOFK FOREIGN KEY (CARD_NO)
	  REFERENCES BORROWER (CARD_NO), 
	 CONSTRAINT ISBN10FK FOREIGN KEY (ISBN10)
	  REFERENCES BOOKS (ISBN10) );

	  
	  
  CREATE TABLE FINES 
   (	LOAN_ID NUMBER(*,0) NOT NULL, 
	FINE_AMT NUMBER(5,2), 
	PAID VARCHAR2(5 BYTE), 
	CONSTRAINT LOAN_IDPK PRIMARY KEY (LOAN_ID),
	 CONSTRAINT FINE_FK FOREIGN KEY (LOAN_ID)
	  REFERENCES BOOK_LOANS (LOAN_ID) 
   ) ;
